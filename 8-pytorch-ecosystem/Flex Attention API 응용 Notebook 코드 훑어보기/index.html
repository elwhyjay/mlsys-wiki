def calculate_tflops(flops: float, time_ms: float, multiplier: int) -> float:
    """
    계산TFLOPS。

    파라미터:
    - flops: 이 부분은 원문의 해당 기술 설명을 이어서 서술한다
    - time_ms: 이 부분은 원문의 해당 기술 설명을 이어서 서술한다
    - multiplier: 관련 내용

    반환한다:
    - TFLOPS관련 내용
    """
    return multiplier * flops * (1e3 / time_ms) / 1e12
